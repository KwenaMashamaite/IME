103  static auto createRandomNumGenerator(int min, int max) {
104  return [distribution = std::uniform_int_distribution(min, max)]() mutable {
105 #if defined(__GNUC__)
106  // MinGW compiler on windows does not provide non-deterministic
107  // values from its std::random_device, so the random numbers
108  // are the same on every run
109  static auto randomEngine = std::mt19937(std::time(nullptr));
110  return distribution(randomEngine);
111 #else
112  static auto randomEngine = std::mt19937(std::random_device{}());
113  return distribution(randomEngine);
114 #endif
115  };
116  }
